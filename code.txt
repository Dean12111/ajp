import javax.swing.*;

public class LoginApp extends JFrame {
    
    private JTextField usernameField;
    private JPasswordField passwordField;
    private JButton loginButton;

    public LoginApp() {
        setTitle("Login Form");
        setSize(300, 150);
        setDefaultCloseOperation(EXIT_ON_CLOSE);

        JPanel panel = new JPanel();
        panel.setLayout(new GridLayout(3, 2,20,20));

        
        JLabel usernameLabel = new JLabel("Username:");
        usernameLabel.setBounds(300, 300, 100, 100);
        
        JLabel passwordLabel = new JLabel("Password:");
        passwordLabel.setBounds(300, 500, 100, 100);
        usernameField = new JTextField();
        usernameField.setBounds(450, 300, 100, 100);
        passwordField = new JPasswordField();
        passwordField.setBounds(450, 500, 100, 100);
        loginButton = new JButton("Login");
        loginButton.setBounds(400, 600, 100, 100);

        panel.add(usernameLabel);
        panel.add(usernameField);
        panel.add(passwordLabel);
        panel.add(passwordField);
        panel.add(new JLabel()); // Empty label for spacing
        panel.add(loginButton);

        loginButton.addActionListener(new ActionListener() {
            @Override
            public void actionPerformed(ActionEvent e) {
                String username = usernameField.getText();
                char[] passwordChars = passwordField.getPassword();
                String password = new String(passwordChars);

                // Replace this with your authentication logic
                if (username.equals("1") && password.equals("1")) {
                    // Open a new window upon successful login
                    openNewWindow();
                } else {
                    JOptionPane.showMessageDialog(LoginApp.this, "Invalid username or password", "Error", JOptionPane.ERROR_MESSAGE);
                }

                // Clear the fields after login attempt
                usernameField.setText("");
                passwordField.setText("");
            }
        });

        add(panel);
        setLocationRelativeTo(null); // Center the window
    }

    // Method to open a new window
    private void openNewWindow() {
        JFrame mainJFrame = new JFrame("Main Window");
        mainJFrame.setSize(1500, 800);
        mainJFrame.setDefaultCloseOperation(JFrame.EXIT_ON_CLOSE);
    
        JPanel buttonPanel = new JPanel(new GridLayout(3, 4)); // 3 rows and 4 columns for the buttons
    
        // Create and add 12 image icon buttons
        for (int i = 1; i <= 12; i++) {
            String imagePath = "/Resources/" + i + ".png"; // Construct the image path

            // Load an ImageIcon from the resource
            ImageIcon icon = new ImageIcon(LoginApp.class.getResource(imagePath));

            JButton button = new JButton(icon);
            final int buttonNumber = i;

            // Add an ActionListener to each button to open a new window
            button.addActionListener(new ActionListener() {
                @Override
                public void actionPerformed(ActionEvent e) {
                    // Call the specific method for the clicked button
                    switch ( buttonNumber) {
                        case 1:
                            openWindowForButton1();
                            break;
                        case 2:
                            openWindowForButton2();
                            break;
                        case 3:
                            openWindowForButton3();
                            break;
                        case 4:
                            openWindowForButton4();
                            break;
                        case 5:
                            openWindowForButton5();
                            break;
                        case 6:
                            openWindowForButton6();
                            break;
                        case 7:
                            openWindowForButton7();
                            break;
                        case 8:
                            openWindowForButton8();
                            break;
                        case 9:
                            openWindowForButton9();
                            break;
                        case 10:
                            openWindowForButton10();
                            break;
                        case 11:
                            openWindowForButton11();
                            break;
                        case 12:
                            openWindowForButton12();
                            break;
                           
                    }
                }
            });

            buttonPanel.add(button);
        }
        
    
        mainJFrame.add(buttonPanel);
        mainJFrame.setLocationRelativeTo(null);
        mainJFrame.setVisible(true);
    }
